def myMethod(datos: List[Double]): Double = {
  val logs: List[Double] = datos.map(Math.log)
  val n = logs.size
  if (n == 0) {
    return 0.0 // O puedes lanzar una excepción o devolver Double.NaN
  }

  val sumaLogs: Double = logs.sum
  val promedioLog: Double = sumaLogs / n // Esto es lo que llamaste 'promediolog2'
  val sumaCuadrados: Double = logs.map { logVal =>
    val diferencia = logVal - promedioLog
    diferencia * diferencia // Elevamos al cuadrado
  }.sum // Sumamos todos los resultados
  Math.sqrt(sumaCuadrados / n)
}
